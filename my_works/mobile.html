rf_out <- randomForest(smartphone ~ ., data=rf_df, ntree=100)

# Extracts variable importance (Mean Decrease in Gini Index)
# Sorts by variable importance and relevels factors to match ordering
var_importance <- data_frame(variable=setdiff(colnames(rf_df), "smartphone"),
                             importance=as.vector(importance(rf_out)))
var_importance <- arrange(var_importance, desc(importance))
var_importance$variable <- factor(var_importance$variable, levels=var_importance$variable)

p <- ggplot(var_importance, aes(x=variable, weight=importance, fill=variable))
p <- p + geom_bar() + ggtitle("Variable Importance from Random Forest Fit")
p <- p + xlab("Demographic Attribute") + ylab("Variable Importance (Mean Decrease in Gini Index)")
p <- p + scale_fill_discrete(name="Variable Name")
p + theme(axis.text.x=element_blank(),
          axis.text.y=element_text(size=12),
          axis.title=element_text(size=16),
          plot.title=element_text(size=18),

legend.text=element_text(size=12))

Для построения этого графика были убраны значения с большим количеством пропусков. Как правило, это характерно для уточняющих вопросов. Так, например, после вопроса о браке идет вопрос о дате регистрации брака. Так случайный лес мы строили по 53 наблюдениям из 72.

Самым информативным признаком оказался tj184, который является ответом на вопрос есть ли у вас мобильный телефон. Причем, корреляция между наличием мобильного телефон и наличием смартфона отрицательная -0.6673951. Это говорит о том, что в нашей выборке количество людей с обычными телефонами больше, чем со смартфонами, так как смартфон является телефоном, но не наоборот.
